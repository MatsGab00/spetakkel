  <script>
    const prompts = [
      "What is your favorite book?",
      "If you could travel anywhere in the world, where would you go?",
      "What is your dream job?",
      "What is your favorite hobby?",
      "If you could have any superpower, what would it be?",
      // Add more prompts as needed
    ];
    
    function generatePromptAndSelectPlayer() {
      const promptText = document.getElementById("prompt-text");
      const selectedPlayerText = document.getElementById("selected-player");
      const selectedPromptText = document.getElementById("selected-prompt");
      
      const playerNamesInput = document.getElementById("player-names");
      const playerNames = playerNamesInput.value.trim();
      
      if (playerNames === "") {
        alert("Please enter at least one player name.");
        return;
      }
      
      const players = playerNames.split(",").map(player => player.trim()).filter(Boolean);
      
      if (players.length === 0) {
        alert("Please enter at least one player name.");
        return;
      }
      
      const randomPlayerIndex = Math.floor(Math.random() * players.length);
      const randomPromptIndex = Math.floor(Math.random() * prompts.length);
      
      const selectedPlayer = players[randomPlayerIndex];
      const selectedPrompt = prompts[randomPromptIndex];
      
      selectedPlayerText.textContent = "Selected Player: " + selectedPlayer;
      selectedPromptText.textContent = "Prompt: " + selectedPrompt;
      promptText.textContent = selectedPrompt;
